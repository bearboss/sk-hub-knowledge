# Jina

```
export PYTHONHOME=/usr/local/python3/
export PYTHONPATH=/usr/local/python3/bin/
export PATH=$PYTHONPATH:$PATH

export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/usr/local/openssl/lib

yum -y install zlib-devel bzip2-devel openssl-devel ncurses-devel sqlite-devel readline-devel tk-devel gdbm-devel db4-devel libpcap-devel xz-devel
yum install -y libffi-devel zlib1g-dev
yum install zlib* -y
yum -y install perl
yum -y install gcc*

wget http://www.openssl.org/source/openssl-1.1.1.tar.gz

./config --prefix=/usr/local/openssl shared zlib

make && make install  

wget https://www.python.org/ftp/python/3.10.0/Python-3.10.0.tgz

./configure --prefix=/usr/local/python3  --enable-optimizations --with-openssl=/usr/local//openssl

make && make install

rm -rf  /usr/bin/python

rm -rf  /usr/bin/pip

# 修复yum

vim /usr/bin/yum
#!/usr/bin/python ==> #!/usr/bin/python2.7

vim /usr/libexec/urlgrabber-ext-down
#!/usr/bin/python ==> #!/usr/bin/python2.7

ln -s /usr/local/python3/bin/python3.10 /usr/bin/python

ln -s /usr/local/python3/bin/pip3.10  /usr/bin/pip

python
import ssl
import _ssl

python -m venv tutorial-env

source tutorial-env/bin/activate

python -m pip install --upgrade pip -i http://pypi.douban.com/simple --trusted-host pypi.douban.com

pip install wheel  -i http://pypi.douban.com/simple --trusted-host pypi.douban.com

pip install jina  -i http://pypi.douban.com/simple --trusted-host pypi.douban.com

jina new hello-jina

cd hello-jina/

jina flow --uses flow.yml

CentOS使用yum指令显示"ImportError:Nomodulenamedsite"错误
unset PYTHONPATH
unset PYTHONHOME
```
